public class Break
{

	static int j = 42;
	
	public void testSwitch(){
mylabel:
		switch(j) {
			case 42:
				return;
			case 43:
				break mylabel;
			case 44:
				break;
		}
	}
	
	public void testWhile(){
mylabel:
		while(true){
			if(j == 1){
				break;
			} else if(j == 2) {
				break mylabel;
			}
			break mylabel;
		}
	}
	
	public void testFor(){
mylabel:
		for (int i = 0; i < 10; i++) {
			if(j == 1){
				break;
			} else if(j == 2) {
				break mylabel;
			}
			break mylabel;
		}
	}

}
