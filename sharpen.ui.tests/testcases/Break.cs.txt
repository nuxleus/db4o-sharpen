public class Break
{
	internal static int j = 42;

	public virtual void testSwitch()
	{
		switch (j)
		{
			case 42:
			{
				return;
			}

			case 43:
			{
				goto mylabel;
			}

			case 44:
			{
				break;
			}
		}
mylabel:
	}

	public virtual void testWhile()
	{
		while (true)
		{
			if (j == 1)
			{
				break;
			}
			else
			{
				if (j == 2)
				{
					goto mylabel;
				}
			}
			goto mylabel;
		}
mylabel:
	}

	public virtual void testFor()
	{
		for (int i = 0; i < 10; i++)
		{
			if (j == 1)
			{
				break;
			}
			else
			{
				if (j == 2)
				{
					goto mylabel;
				}
			}
			goto mylabel;
		}
mylabel:
	}

	public virtual void testNestedLoops()
	{
		for (int i = 0; i < 10; i++)
		{
			for (int k = 0; k < 10; k++)
			{
				for (int m = 0; m < 10; m++)
				{
					if (i == 5 && k == 5 && m == 5)
					{
						goto mylabel1;
					}
					else
					{
						goto mylabel2;
					}
				}
			}
mylabel2:
		}
mylabel1:
	}
}
